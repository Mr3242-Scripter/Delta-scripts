-- FUNCTIONS

local function addRecent(pos)
	table.insert(recentTeleports,pos)
	local entry = Instance.new("Frame")
	entry.Size = UDim2.new(0,160,0,100)
	entry.BackgroundColor3 = Color3.fromRGB(100,100,255)
	entry.Parent = scrollFrame

	local lbl = Instance.new("TextLabel")
	lbl.Size = UDim2.new(1,-60,0,50)
	lbl.Position = UDim2.new(0,5,0,5)
	lbl.Text = math.floor(pos.X)..","..math.floor(pos.Y)..","..math.floor(pos.Z)
	lbl.TextScaled = true
	lbl.BackgroundTransparency = 1
	lbl.TextColor3 = Color3.new(1,1,1)
	lbl.Parent = entry

	local tpBtn = Instance.new("TextButton")
	tpBtn.Size = UDim2.new(0,50,0,50)
	tpBtn.Position = UDim2.new(1,-55,0,5)
	tpBtn.Text = "TP"
	tpBtn.TextScaled = true
	tpBtn.BackgroundColor3 = Color3.fromRGB(0,200,100)
	tpBtn.TextColor3 = Color3.new(1,1,1)
	tpBtn.Parent = entry

	tpBtn.MouseButton1Click:Connect(function()
		local char = player.Character
		if char and char:FindFirstChild("HumanoidRootPart") then
			char.HumanoidRootPart.CFrame = CFrame.new(pos)
		end
	end)

	scrollFrame.CanvasSize = UDim2.new(0,layout.AbsoluteContentSize.X,0,scrollFrame.CanvasSize.Y.Offset)
end

-- Teleport typed coordinates
tpButton.MouseButton1Click:Connect(function()
	local text = box.Text
	local split = string.split(text,",")
	if #split == 3 then
		local x = tonumber(split[1])
		local y = tonumber(split[2])
		local z = tonumber(split[3])
		if x and y and z then
			local char = player.Character
			if char and char:FindFirstChild("HumanoidRootPart") then
				local pos = Vector3.new(x,y,z)
				char.HumanoidRootPart.CFrame = CFrame.new(pos)
				lastPosition = pos
				addRecent(pos)
			end
		end
	end
end)

-- Copy position
copyButton.MouseButton1Click:Connect(function()
	local char = player.Character
	if char and char:FindFirstChild("HumanoidRootPart") then
		local pos = char.HumanoidRootPart.Position
		box.Text = math.floor(pos.X)..","..math.floor(pos.Y)..","..math.floor(pos.Z)
	end
end)

-- Movement lock
lockButton.MouseButton1Click:Connect(function()
	dragEnabled = not dragEnabled
	if dragEnabled then
		lockButton.Text = "Unlock Movement"
	else
		lockButton.Text = "Lock Movement"
	end
end)

-- TP-to-Mouse toggle
mouseButton.MouseButton1Click:Connect(function()
	tpMouseEnabled = not tpMouseEnabled
	if tpMouseEnabled then
		mouseButton.Text = "TP To Mouse: ON"
		mouseButton.BackgroundColor3 = Color3.fromRGB(0,220,120)
	else
		mouseButton.Text = "TP To Mouse: OFF"
		mouseButton.BackgroundColor3 = Color3.fromRGB(255,0,100)
	end
end)

tpMouseConnection = mouse.Button1Down:Connect(function()
	if tpMouseEnabled then
		local char = player.Character
		if char and char:FindFirstChild("HumanoidRootPart") then
			local pos = mouse.Hit.Position
			char.HumanoidRootPart.CFrame = CFrame.new(pos + Vector3.new(0,5,0))
		end
	end
end)

-- Show/hide recent list
recentButton.MouseButton1Click:Connect(function()
	listGui.Visible = not listGui.Visible
end)

closeList.MouseButton1Click:Connect(function()
	listGui.Visible = false
end)

-- Live position display
task.spawn(function()
	while gui.Parent do
		task.wait(0.2)
		local char = player.Character
		if char and char:FindFirstChild("HumanoidRootPart") then
			local pos = char.HumanoidRootPart.Position
			posLabel.Text = "Position: "..math.floor(pos.X)..","..math.floor(pos.Y)..","..math.floor(pos.Z)
		end
	end
end)

-- Destroy GUI
destroyButton.MouseButton1Click:Connect(function()
	if tpMouseConnection then tpMouseConnection:Disconnect() end
	gui:Destroy()
end)

-- DRAGGING FIXED (top bar)
topBar.InputBegan:Connect(function(input)
	if (input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch) and dragEnabled then
		dragging = true
		dragStart = input.Position
		startPos = frame.Position
		input.Changed:Connect(function()
			if input.UserInputState == Enum.UserInputState.End then
				dragging = false
			end
		end)
	end
end)

topBar.InputChanged:Connect(function(input)
	if dragging and (input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch) then
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X,
								   startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
end)
